Code to see if a word is a palindrome

Sep 2, 2022 â€¢ 1 min read

def isPalindrome(s):
	print(s)
	print(s[::-1])
	return s == s[::-1]

s = input("Enter a word: ")
ans = isPalindrome(s)

if ans:
	print(s + " is a palindrome")
else:
	print(s + " is not a palindrome")
